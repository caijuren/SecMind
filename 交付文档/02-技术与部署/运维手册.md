## 2. 服务启停操作指南

### 2.1 Docker Compose 环境

#### 启动全部服务
```bash
# 进入部署目录
cd /opt/secmind

# 启动全部服务
docker compose -f docker-compose.prod.yml up -d

# 验证服务状态
docker compose -f docker-compose.prod.yml ps

# 查看启动日志
docker compose -f docker-compose.prod.yml logs --tail=100 -f
```

#### 停止全部服务(计划内停机)
```bash
# 优雅停止（先停止API流量，再停止服务）
# 1. 从Nginx摘除节点（如果是多节点部署时）
ssh nginx-lb "sed -i 's/server api-node-1:8000/# server api-node-1:8000/' /etc/nginx/conf.d/secmind.conf && nginx -s reload"

# 2. 等待正在处理的请求完成（监控连接数归零）
sleep 30

# 3. 停止容器（保留数据卷）
docker compose -f docker-compose.prod.yml down

# 4. 如需同时清理数据卷（谨慎操作）
docker compose -f docker-compose.prod.yml down -v
```

#### 重启单个服务
```bash
# 重启 API 服务（滚动更新）
docker compose -f docker-compose.prod.yml up -d --no-deps --scale api-server=0 api-server
sleep 5
docker compose -f docker-compose.prod.yml up -d --no-deps --scale api-server=4 api-server

# 重启单个容器
docker compose -f docker-compose.prod.yml restart api-server

# 重新构建并启动
docker compose -f docker-compose.prod.yml up -d --build --no-deps api-server
```

#### 查看服务状态
```bash
# 查看所有容器状态
docker compose -f docker-compose.prod.yml ps

# 查看服务资源占用
docker stats --no-stream

# 查看各容器日志
docker compose -f docker-compose.prod.yml logs -f --tail=50 api-server
docker compose -f docker-compose.prod.yml logs -f --tail=50 postgres-primary
docker compose logs postgres-primary --tail=50 |grep -i error

# 特定时间范围日志
docker compose logs --since "2025-01-01T10:00:00" --until "2025-01-01T12:00:00" api-server
```

## 4. 数据库运维

### 4.1 数据库连接与基础操作

```bash
# 通过 Docker 连接数据库
docker exec -it postgres-primary psql -U secmind -d secmind

# 查看数据库连接数
SELECT datname, numbackends, active_queries FROM pg_stat_database;

# 查看当前活跃查询
SELECT pid, usename, application_name, client_addr, state, query_start, query
FROM pg_stat_activity
WHERE state != 'idle'
ORDER BY query_start DESC;

# 终止特定查询
SELECT pg_terminate_backend(pid)
FROM pg_stat_activity
WHERE pid = 12345;

# 查看表大小排行
SELECT
    relname AS table_name,
    pg_size_pretty(pg_total_relation_size(relid)) AS total_size,
    pg_size_pretty(pg_relation_size(relid)) AS data_size,
    pg_size_pretty(pg_total_relation_size(relid) - pg_relation_size(relid)) AS index_size
FROM pg_catalog.pg_statio_user_tables
ORDER BY pg_total_relation_size(relid) DESC
LIMIT 20;

# 查看数据库大小
SELECT
    datname,
    pg_size_pretty(pg_database_size(datname)) AS db_size
FROM pg_database
ORDER BY pg_database_size(datname) DESC;
```

### 4.2 备份操作

#### 全量备份
```bash
#!/bin/bash
# /usr/local/bin/backup_secmind_full.sh
# 每日凌晨2:00 执行

BACKUP_DIR=/data/backups/postgres
DATE=$(date +%Y%m%d_%H%M%S)
DB_NAME=secmind
DB_USER=backup_user
DB_HOST=10.0.1.10
RETENTION_DAYS=30
MINIO_BUCKET=secmind-backups

# 创建备份目录
mkdir -p ${BACKUP_DIR}

# 执行 pg_basebackup (物理全量备份，包含WAL)
pg_basebackup -h ${DB_HOST} -D ${BACKUP_DIR}/full_${DATE} \
  -U ${DB_USER} -P -v --wal-method=stream \
  --label="secmind_full_backup_$(date +%Y%m%d)"

# 压缩
cd ${BACKUP_DIR}
tar -czf full_${DATE}.tar.gz full_${DATE}
rm -rf full_${DATE}

# 上传到 MinIO
mc cp full_${DATE}.tar.gz minio/${MINIO_BUCKET}/full_backups/

# 清理本地过期备份
find ${BACKUP_DIR} -name "*.tar.gz" -type f -mtime +${RETENTION_DAYS} -delete

# 清理 MinIO 过期备份
mc rm --recursive --older-than 30d minio/${MINIO_BUCKET}/full_backups/

# 记录备份日志
echo "[$(date)] 全量备份完成: full_${DATE}.tar.gz" >> /var/log/backup.log

# 验证备份完整性
tar -tzf full_${DATE}.tar.gz > /dev/null 2>&1
if [ $? -eq 0 ]; then
    echo "[$(date)] 备份完整性验证通过" >> /var/log/backup.log
else
    echo "[$(date)] 备份完整性验证失败" >> /var/log/backup.log
    # 发送告警
    curl -X POST -H "Content-Type: application/json" \
      -d '{"msgtype": "text", "text": {"content": "SecMind备份完整性验证失败"}}' \
      https://qyapi.weixin.qq.com/cgi-bin/webhook/send?key=${WEBHOOK_KEY}
fi
```

#### 逻辑备份（按Schema）
```bash
#!/bin/bash
# /usr/local/bin/backup_secmind_logic.sh
# 每周日凌晨4:00 执行

DATE=$(date +%Y%m%d)
DB_HOST=10.0.1.10
DB_USER=backup_user
DB_NAME=secmind
BACKUP_DIR=/data/backups/logical

# 全库逻辑备份（压缩格式）
pg_dump -h ${DB_HOST} -U ${DB_USER} -d ${DB_NAME} \
  -F c -f ${BACKUP_DIR}/secmind_full_${DATE}.dump \
  -v

# 仅备份Schema结构
pg_dump -h ${DB_HOST} -U ${DB_USER} -d ${DB_NAME} \
  --schema-only -f ${BACKUP_DIR}/secmind_schema_${DATE}.sql \
  -v

# 仅备份数据（无索引/约束，加速恢复）
pg_dump -h ${DB_HOST} -U ${DB_USER} -d ${DB_NAME} \
  --data-only --exclude-table-data='raw_logs' \
  -f ${BACKUP_DIR}/secmind_data_${DATE}.sql \
  -v

# 上传到 MinIO 异地
mc cp ${BACKUP_DIR}/secmind_full_${DATE}.dump \
  minio/secmind-backups/logical_backups/

# 保留90天
find ${BACKUP_DIR} -name "*.dump" -type f -mtime +90 -delete
find ${BACKUP_DIR} -name "*.sql" -type f -mtime +90 -delete
```

### 4.3 恢复操作

#### 从物理备份恢复
```bash
#!/bin/bash
# /usr/local/bin/restore_secmind_full.sh
# 使用前请确认：

RESTORE_DATE=$1  # 格式: YYYYMMDD
BACKUP_FILE=/data/backups/postgres/full_${RESTORE_DATE}.tar.gz

if [ ! -f "${BACKUP_FILE}" ]; then
    # 从 MinIO 下载
    mc cp minio/secmind-backups/full_backups/full_${RESTORE_DATE}.tar.gz ${BACKUP_FILE}
fi

# 1. 停止目标数据库
systemctl stop postgresql

# 2. 清理旧数据目录
mv /var/lib/postgresql/15/data /var/lib/postgresql/15/data_old

# 3. 解压备份
mkdir -p /var/lib/postgresql/15/data
tar -xzf ${BACKUP_FILE} -C /var/lib/postgresql/15/data/

# 4. 修改配置文件
cat > /var/lib/postgresql/15/data/postgresql.conf << 'EOF'
# 默认配置，根据环境调整
listen_addresses = '*'
port = 5432
max_connections = 200
shared_buffers = 16GB
EOF

# 5. 启动数据库
systemctl start postgresql

# 6. 验证数据
psql -c "SELECT count(*) FROM users;"
psql -c "SELECT count(*) FROM alerts;"

# 7. 执行 VACUUM ANALYZE
psql -c "VACUUM ANALYZE;"
```

#### 时间点恢复 (PITR)
```bash
# 时间点恢复场景：恢复到 2025-01-01 14:30:00 的状态

# 1. 准备恢复环境
mkdir -p /var/lib/postgresql/15/data_pitr
cd /var/lib/postgresql/15/data_pitr

# 2. 从 MinIO 下载基础备份
mc cp --recursive minio/secmind-backups/full_backups/full_20250101/ ./

# 3. 配置恢复参数
cat > recovery.conf << 'EOF'
restore_command = 'mc cp minio/secmind-backups/wal/%f %p'
recovery_target_time = '2025-01-01 14:30:00 CST'
recovery_target_action = 'pause'
EOF

# 4. 启动 PostgreSQL 进入恢复模式
pg_ctl -D /var/lib/postgresql/15/data_pitr start

# 5. 监控恢复进度
psql -c "SELECT pg_is_in_recovery();"
psql -c "SELECT pg_last_wal_receive_lsn(), pg_last_wal_replay_lsn();"

# 6. 到达目标时间后，确认数据
psql -c "SELECT count(*) FROM alerts WHERE created_at < '2025-01-01 14:30:00';"

# 7. 完成恢复
psql -c "SELECT pg_wal_recovery_finish();"
```

### 4.5 数据库性能优化

```sql
-- 查看需要VACUUM的表
SELECT
    relname,
    n_dead_tup,
    n_live_tup,
    round(n_dead_tup * 100.0 / (n_live_tup + n_dead_tup + 1), 2) AS dead_pct
FROM pg_stat_user_tables
ORDER BY n_dead_tup DESC
LIMIT 10;

-- 手动执行 VACUUM
VACUUM (VERBOSE, ANALYZE) alerts;

-- 重建索引
REINDEX INDEX idx_alerts_created_at;
REINDEX TABLE alerts;

-- 查看索引使用情况
SELECT
    schemaname,
    tablename,
    indexname,
    idx_scan,
    idx_tup_read,
    idx_tup_fetch
FROM pg_stat_user_indexes
ORDER BY idx_scan ASC
LIMIT 20;  -- 未使用的索引

-- 查看表膨胀率
SELECT
    schemaname || '.' || relname AS table_name,
    pg_size_pretty(pg_relation_size(relid)) AS table_size,
    round(100 * (n_dead_tup::float8 / GREATEST(n_live_tup + n_dead_tup, 1)), 2) AS dead_pct
FROM pg_stat_user_tables
WHERE n_dead_tup > 10000
ORDER BY dead_pct DESC;
```

### 6.3 数据归档

```bash
#!/bin/bash
# /usr/local/bin/archive_old_data.sh
# 每月1日执行，归档超过180天的数据

ARCHIVE_DIR=/data/archives
RETENTION_MONTHS=6
DB_NAME=secmind

# 1. 归档过期告警数据到文件

    COPY (
        SELECT * FROM alerts
        WHERE created_at < NOW() - INTERVAL '180 days'
        ORDER BY created_at
    ) TO '${ARCHIVE_DIR}/alerts_archive_$(date +%Y%m).csv' WITH CSV HEADER;
"

# 2. 从主表删除已归档数据（分批次删除避免长时间锁表）
psql -h localhost -U secmind -d ${DB_NAME} << 'SQL'
DO $$
DECLARE
    deleted_rows INTEGER;
BEGIN
    LOOP
        DELETE FROM alerts
        WHERE created_at < NOW() - INTERVAL '180 days'
        AND id IN (
            SELECT id FROM alerts
            WHERE created_at < NOW() - INTERVAL '180 days'
            LIMIT 10000
        );

        GET DIAGNOSTICS deleted_rows = ROW_COUNT;

        IF deleted_rows = 0 THEN
            EXIT;
        END IF;

        COMMIT;
        PERFORM pg_sleep(1);
    END LOOP;
END $$;
SQL

# 3. 归档文件压缩
gzip ${ARCHIVE_DIR}/alerts_archive_$(date +%Y%m).csv

# 4. 上传到 MinIO 冷存储
mc cp ${ARCHIVE_DIR}/alerts_archive_$(date +%Y%m).csv.gz \
  minio/secmind-cold-storage/alerts/

# 5. VACUUM 释放空间
psql -h localhost -U secmind -d ${DB_NAME} -c "VACUUM VERBOSE ANALYZE alerts;"

# 6. 重建索引
psql -h localhost -U secmind -d ${DB_NAME} -c "REINDEX TABLE alerts;"
```

## 8. 常见故障处理流程

### 8.1 服务宕机

#### 故障现象
- 用户无法访问平台
- 健康检查返回非200状态码
- 监控告警触发

#### 处理步骤

```
步骤1: 确认故障范围
  -> curl -k https://localhost/api/v1/health
  -> docker ps -a | grep secmind
  -> systemctl status docker

步骤2: 查看错误日志
  -> docker logs --tail=200 api-server
  -> tail -200 /var/log/secmind/api/error.log

步骤3: 重启服务
  -> docker compose restart api-server
  -> 或 kubectl rollout restart deployment api-server

步骤4: 若无法恢复，回滚至上一版本
  -> docker compose down
  -> git checkout <previous_tag>
  -> docker compose up -d

步骤5: 事后分析
  -> 收集日志和dump
  -> 分析根因
  -> 更新运维文档
```

#### 快速恢复脚本
```bash
#!/bin/bash
# /usr/local/bin/quick_recovery.sh
# 快速恢复服务

echo "[$(date)] 开始服务恢复..."

# 1. 检查 Docker 服务
if ! systemctl is-active --quiet docker; then
    echo "Docker not running, restarting..."
    systemctl restart docker
    sleep 10
fi

# 2. 检查数据库
docker exec postgres-primary pg_isready -U secmind
if [ $? -ne 0 ]; then
    echo "PostgreSQL 异常，尝试重启..."
    docker compose restart postgres-primary
    sleep 20
fi

# 3. 检查 Redis
docker exec redis-master redis-cli ping
if [ $? -ne 0 ]; then
    echo "Redis 异常，尝试重启..."
    docker compose restart redis-master
    sleep 5
fi

# 4. 启动所有服务
docker compose -f docker-compose.prod.yml up -d

# 5. 等待健康检查通过
for i in {1..30}; do
    STATUS=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:8000/api/v1/health 2>/dev/null)
    if [ "$STATUS" == "200" ]; then
        echo "服务恢复成功"
        exit 0
    fi
    echo "等待服务启动... ($i/30)"
    sleep 5
done

echo "服务恢复完成，请手动检查"
```

### 8.2 数据库连接池满

#### 故障现象
- API 日志频繁出现 `psycopg2.OperationalError: FATAL: remaining connection slots are reserved`
- Grafana 面板显示连接数达到上限
- 新请求超时失败

#### 处理步骤

```bash
# 步骤1: 立即释放异常连接
psql -c "
    SELECT pg_terminate_backend(pid)
    FROM pg_stat_activity
    WHERE state = 'idle in transaction'
    AND state_change < NOW() - INTERVAL '5 minutes';
"

# 步骤2: 查看当前连接分布
psql -c "
    SELECT
        state,
        usename,
        count(*) as connections,
        application_name
    FROM pg_stat_activity
    GROUP BY state, usename, application_name
    ORDER BY connections DESC;
"

# 步骤3: 临时增加最大连接数
psql -c "ALTER SYSTEM SET max_connections = 300;"
psql -c "SELECT pg_reload_conf();"

# 步骤4: 检查慢查询并终止
psql -c "
    SELECT pid, now() - pg_stat_activity.query_start AS duration,
           query, state
    FROM pg_stat_activity
    WHERE state != 'idle'
    ORDER BY duration DESC
    LIMIT 10;
"

# 步骤5: 优化连接池配置
# 在 .env 文件中调整
DATABASE_POOL_SIZE=30
DATABASE_MAX_OVERFLOW=30
DATABASE_POOL_TIMEOUT=60

# 步骤6: 重启 API 服务使配置生效
docker compose restart api-server
```

- Nginx 日志出现 `upstream prematurely closed connection`
- API 服务可能仍在运行

### 9.1 全量备份流程

```
时间: 每日凌晨 02:00 - 03:00
方式: pg_basebackup 物理全量备份
存储: 本地保留30天 + MinIO对象存储

备份流程时序:
02:00  - 开始 pg_basebackup
02:30  - 备份完成，开始压缩
02:35  - 压缩完成，上传至 MinIO
02:40  - 本地清理过期备份
02:45  - 验证备份完整性
02:50  - 发送备份状态通知
```

#### 完整备份脚本
```bash
#!/bin/bash
# /usr/local/bin/complete_backup.sh
set -e

# === 配置 ===
DB_HOST="10.0.1.10"
DB_PORT="5432"
DB_NAME="secmind"
DB_USER="backup_user"
BACKUP_BASE="/data/backups"
MINIO_BUCKET="secmind-backups"
RETENTION_DAYS=30
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="${BACKUP_BASE}/full_${DATE}"
LOG_FILE="/var/log/backup_${DATE}"

# === 初始化 ===
mkdir -p ${BACKUP_DIR}
exec 2>> ${LOG_FILE}
echo "[$(date)] === 开始全量备份 ==="

# === 步骤1: PostgreSQL 全量备份 ===
echo "[$(date)] 开始 pg_basebackup..."
pg_basebackup -h ${DB_HOST} -p ${DB_PORT} -D ${BACKUP_DIR} \
  -U ${DB_USER} -P -v --wal-method=stream \
  --label="secmind_auto_backup_${DATE}"

# === 步骤2: 配置文件和密钥备份 ===
echo "[$(date) ] 备份配置文件..."
tar -czf ${BACKUP_DIR}/config_backup.tar.gz \
  /opt/secmind/.env \
  /opt/secmind/nginx/ssl/ \
  /opt/secmind/docker-compose*.yml

# === 步骤3: 压缩 ===
echo "[$(date)] 压缩备份..."
cd ${BACKUP_BASE}/ && tar -czf full_${DATE}.tar.gz $(basename ${BACKUP_DIR})
rm -rf ${BACKUP_DIR}

# === 步骤4: 上传至MinIO ===
echo "[$(date)] 上传至 MinIO..."
mc cp full_${DATE}.tar.gz minio/${MINIO_BUCKET}/full_backups/

# === 步骤5: 验证 ===
echo "[$(date)] 验证备份完整性..."
md5sum full_${DATE}.tar.gz > full_${DATE}.tar.gz.md5
mc cp full_${DATE}.tar.gz.md5 minio/${MINIO_BUCKET}/full_backups/

# === 步骤6: 清理本地备份 ===
echo "[$(date)] 清理本地过期备份..."
find ${BACKUP_BASE} -name "*.tar.gz" -type f -mtime +${RETENTION_DAYS} -delete
find ${BACKUP_BASE} -name "*.md5" -type f -mtime +${RETENTION_DAYS} -delete

# === 步骤7: 清理MinIO过期备份 ===
echo "[$(date)] 清理 MinIO 过期备份..."
mc rm --recursive --older-than ${RETENTION_DAYS}d \
  minio/${MINIO_BUCKET}/full_backups/ --force

# === 完成 ===
echo "[$(date)] === 全量备份完成 ==="

# 发送通知（可选）
BACKUP_SIZE=$(ls -lh full_${DATE}.tar.gz | awk '{print $5}')
curl -s -X POST "https://qyapi.weixin.qq.com/cgi-bin/webhook/send?key=${WEBHOOK_KEY}" \
  -H "Content-Type: application/json" \
  -d "{
    \"msgtype\": \"markdown\",
    \"markdown\": {
      \"content\": \"**SecMind 备份完成通知**\n> 备份时间: ${DATE}\n> 备份大小: ${BACKUP_SIZE}\n> 保留天数: ${RETENTION_DAYS}天\n> 存储位置: MinIO/${MINIO_BUCKET}\"
    }
  }"
```

### 9.3 完整恢复流程

```bash
#!/bin/bash
# /usr/local/bin/disaster_recovery.sh
# 灾难恢复脚本 - 从零恢复完整恢复

RESTORE_DATE="${1:-latest}"   # 格式: YYYYMMDD 或 latest
RESTORE_TARGET_TIME="$2"      # PITR目标时间，可选

echo "[$(date)] === 开始灾难恢复 ==="

# === 阶段1: 基础设施准备 ===
echo "[$(date)] 阶段1: 准备基础设施..."

# 停止现有服务
docker compose down

# 清理旧数据
rm -rf /var/lib/postgresql/15/data/*
rm -rf /var/lib/redis/data/*

# 从MinIO拉取备份
echo "[$(date)] 从MinIO拉取备份..."
if [ "${RESTORE_DATE}" == "latest" ]; then
    BACKUP_FILE=$(mc ls minio/secmind-backups/full_backups/ | sort | tail -1 | awk '{print $NF}')
else
    BACKUP_FILE="full_${RESTORE_DATE}.tar.gz"
fi

mc cp minio/secmind-backups/full_backups/${BACKUP_FILE} /tmp/

# 验证MD5
mc cp minio/secmind-backups/full_backups/${BACKUP_FILE}.md5 /tmp/
cd /tmp && md5sum -c ${BACKUP_FILE}.md5
if [ $? -ne 0 ]; then
    echo "备份文件完整性校验失败，终止恢复"
    exit 1
fi

# === 阶段2: 恢复PostgreSQL ===
echo "[$(date)] 阶段2: 恢复 PostgreSQL..."

# 解压备份
tar -xzf /tmp/${BACKUP_FILE} -C /tmp/
FULL_DIR=$(find /tmp -maxdepth 1 -type d -name "full_*" | head -1)

# 复制数据文件
cp -r ${FULL_DIR}/base/* /var/lib/postgresql/15/data/

# 如果是 PITR，配置恢复目标
if [ ! -z "${RESTORE_TARGET_TIME}" ]; then
    cat > /var/lib/postgresql/15/data/recovery.signal << 'EOF'
EOF

    cat > /var/lib/postgresql/15/data/postgresql.conf << 'EOF'
restore_command = 'mc cat minio/secmind-backups/wal/%f.gz | gunzip -c > %p'
recovery_target_time = '${RESTORE_TARGET_TIME}'
recovery_target_action = 'promote'
EOF
fi
EOF
    echo "配置为时间点恢复: ${RESTORE_TARGET_TIME}"
fi

# 设置权限
chown -R postgres:postgres /var/lib/postgresql/15/data/
chmod 700 /var/lib/postgresql/15/data/

# === 阶段3: 启动数据库 ===
echo "[$(date)] 阶段3: 启动数据库..."
docker compose up -d postgres-primary

# 等待数据库就绪
for i in {1..60}; do
    sleep 5
    docker exec postgres-primary pg_isready -U secmind && break
done

# === 阶段4: 恢复Redis ===
echo "[$(date)] 阶段4: 恢复Redis..."
docker compose up -d redis-master

# Redis 从备份恢复（如果有持久化文件）
# 默认Redis只做缓存，可直接从数据库重建

# === 阶段5: 启动其他服务 ===
echo "[$(date)]: 启动其他服务..."
docker compose up -d

# === 阶段6: 验证数据 ===
echo "[$(date)] 阶段6: 验证数据..."
docker exec postgres-primary psql -U secmind -d ${DB_NAME} -c "
    SELECT 'users_count', count(*) FROM users UNION ALL
    SELECT 'alerts_count', count(*) FROM alerts UNION ALL
    SELECT 'cases_count', count(*) FROM cases UNION ALL
    SELECT 'tenant_count', count(*) FROM tenants;
"

# === 阶段7: 执行 VACUUM ===
docker exec postgres-primary psql -U secmind -d ${DB_NAME} -c "VACUUM VERBOSE ANALYZE;"

echo "[$(date)] === 灾难恢复完成 ==="
```

### Redis安全加固
- 设置requirepass强密码
- 禁用FLUSHALL/FLUSHDB等危险命令（rename-command）
- 绑定内网IP，禁止公网访问
- 运行非root用户
- 启用TLS加密传输
- 限制内存使用maxmemory，配置淘汰策略

### Docker安全加固
- 使用非root用户运行容器
- 限制容器资源使用（--memory、--cpus）
- 定期扫描镜像漏洞
- 使用私有镜像仓库
- 启用Docker内容信任（DCT）
- 限制容器网络能力（--cap-drop ALL）

### 应用层安全加固
- API接口限流配置
- JWT Token过期时间设置（建议24小时内）
- 敏感配置信息使用环境变量，不硬编码
- 启用审计日志记录所有管理操作
- 定期轮换API Key和密钥
- 会话超时设置（建议30分钟无操作自动退出）

### 定期安全巡检清单
| 周期 | 巡检项目 | 检查方法 |
|------|---------|---------|
| 每日 | 系统资源监控 | top/htop/dstat |
| 每日 | 关键服务状态 | docker ps / systemctl status |
| 每日 | 磁盘使用率 | df -h |
| 每周 | 安全日志检查 | grep -i "fail\|error\|denied" /var/log/secure |
| 每周 | 用户账户审计 | cat /etc/passwd /etc/shadow |
| 每月 | 漏洞扫描 | 使用漏洞扫描工具 |
| 每月 | 备份恢复演练 | 执行一次完整恢复测试 |
| 每季度 | 渗透测试 | 由第三方安全团队执行 |
